        def []=(i, j, v)
                case i
                when Range
                        if i.entries.size == 1
                                self[i.begin, j] = (v.is_a?(Matrix) ? v.row(0) : v)
                        else
                                case j
                                when Range
                                        if j.entries.size == 1
                                                self[i, j.begin] = (v.is_a?(Matrix) ? v.column(0) : v)
                                        else
                                                i.each{|l| self.row= l, v.row(l - i.begin), j}
                                        end
                                else
                                        self.column= j, v, i       
                                end
                        end
                else
                        case j
                        when Range
                                if j.entries.size == 1
                                        self[i, j.begin] = (v.is_a?(Vector) ? v[0] : v)
                                else
                                        self.row= i, v, j
                                end
                        else
                                @rows[i][j] = (v.is_a?(Vector) ? v[0] : v)

                        end
                end           
        end
