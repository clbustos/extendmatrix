        def realSchur(eps = 1.0e-10, steps = 100)
                h = self.hessenberg_form_H
                h1 = Matrix[]
                i = 0
                loop do
                        h1 = h.hessenbergR * h.hessenbergQ
                        break if Matrix.diag_in_delta?(h1, h, eps) or steps <= 0
                        h = h1.clone 
                        steps -= 1
                        i += 1
                end
                h1
        end
