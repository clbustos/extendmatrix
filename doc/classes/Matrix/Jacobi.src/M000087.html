                def Jacobi.max(a)
                        n = a.row_size
                        max = 0
                        p = 0
                        q = 0
                        n.times{|i|
                                ((i+1)...n).each{|j| 
                                        val = a[i, j].abs
                                        if val > max
                                                max = val
                                                p = i
                                                q = j
                                        end        }}
                        return p, q
                end
