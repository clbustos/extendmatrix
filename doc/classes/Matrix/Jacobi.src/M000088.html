                def Jacobi.sym_schur2(a, p, q)
                        if a[p, q] != 0
                                tau = Float(a[q, q] - a[p, p])/(2 * a[p, q])
                                if tau >= 0
                                        t = 1./(tau + Math.sqrt(1 + tau ** 2))
                                else        
                                        t = -1./(-tau + Math.sqrt(1 + tau ** 2))
                                end
                                c = 1./Math.sqrt(1 + t ** 2)
                                s = t * c
                        else
                                c = 1
                                s = 0
                        end
                        return c, s
                end
